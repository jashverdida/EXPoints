// Firestore Security Rules for EXPoints (Production Mode)
// Secure rules that allow public browsing but require auth for interactions

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Users collection - users can only access their own data
    match /users/{userId} {
      allow read: if request.auth != null && request.auth.uid == userId;
      allow write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Reviews collection - public read, authenticated write
    match /reviews/{reviewId} {
      allow read: if true; // Public read access (guests can browse)
      allow create: if request.auth != null && 
                       request.auth.uid == request.resource.data.userId &&
                       isValidReview(request.resource.data);
      allow update: if request.auth != null && 
                       (request.auth.uid == resource.data.userId || // Owner can edit
                        isLikeUpdate(request.resource.data, resource.data)); // Anyone can like
      allow delete: if request.auth != null && request.auth.uid == resource.data.userId;
    }
    
    // Comments collection - public read, authenticated write
    match /comments/{commentId} {
      allow read: if true; // Public read access
      allow create: if request.auth != null && 
                       request.auth.uid == request.resource.data.userId &&
                       isValidComment(request.resource.data);
      allow update, delete: if request.auth != null && request.auth.uid == resource.data.userId;
    }
    
    // Likes collection - authenticated users only
    match /likes/{likeId} {
      allow read, write: if request.auth != null;
      allow create: if request.auth != null && 
                       request.auth.uid == request.resource.data.userId;
    }
    
    // Helper functions for validation
    function isValidReview(data) {
      return data.keys().hasAll(['userId', 'gameTitle', 'rating', 'content']) &&
             data.rating is int &&
             data.rating >= 1 && data.rating <= 5 &&
             data.gameTitle is string && data.gameTitle.size() > 0 &&
             data.content is string && data.content.size() > 0;
    }
    
    function isValidComment(data) {
      return data.keys().hasAll(['userId', 'reviewId', 'content']) &&
             data.content is string && data.content.size() > 0;
    }
    
    function isLikeUpdate(newData, oldData) {
      return newData.diff(oldData).affectedKeys().hasOnly(['likes']) &&
             newData.likes == oldData.likes + 1;
    }
  }
}
